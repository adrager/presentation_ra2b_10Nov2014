\section{Lepton Veto}
\subsection{Introduction}
\begin{frame}
\frametitle{Classical Lepton Veto \& Isolated Track}
  \begin{columns}
    \begin{column}{0.5\textwidth}
     \begin{itemize}
       \item Muon reco \& isolation
       \begin{itemize}
       \item POG recommended \\`Tight Muon`
       \item $p_{T}>10 \gev$ \& $|\eta|<2.5$
       \item Isolation $\text{Iso}_{\text{Rel}}<20\%$ ($\Delta R=0.4$) 
       \end{itemize}
       \item Electron reco \& isolation
       \begin{itemize}
       \item POG recommended \\workpoint `Veto`
       \item $p_{T}>10 \gev$ \& $|\eta|<2.5$
       \item Isolation $\text{Iso}_{\text{Rel}}<20\%$ ($\Delta R=0.3$) 
       \end{itemize}
     \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
     \begin{itemize}
      \item Isolated Track
      \begin{itemize}
       \item Consider charged PF cand with $\text{dz}<0.05$
       \item Sum tracks (Goodvertix collection) $\text{dz}<0.05$
       \item $p_{T}>15 \gev$ and
       \item $\text{Iso}_{\text{cut}}=0.1$ to PF cand
      \end{itemize}
     \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Conclusion}
\begin{frame}
 \begin{itemize}
  \item Isolated Track shows better performance than isolated muon and electron veto
  \item Lepton Veto rejection:$46167.5$ (QCD false rejection: $31.3 $)
  \item Iso. Track rejection:$56369,6$ false $3475.5$ (6\%) (QCD false rejection: $84.8 $)
  \item Purity at around 6\%
  \item Estimatino method for both lost lepton and had tau would need to be modified significantly
 \end{itemize}

\end{frame}
